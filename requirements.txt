# ===== CORE TTS ENGINE =====
# Coqui TTS - main voice cloning engine
coqui-tts

soundfile            # I/O for audio files (WAV, FLAC)  
pydub               # Simple audio manipulation and concatenation
audioread           # Backend for audio file reading
ffmpeg-python       # FFmpeg wrapper for audio conversions (optional)

# ===== DEEP LEARNING & ML =====
# PyTorch ecosystem for XTTS model inference
torch               # Main ML framework with MPS support
torchaudio          # Audio extensions for PyTorch  
transformers        # Hugging Face transformers for NLP
tokenizers          # Fast tokenization for transformers

# ===== NUMERICAL COMPUTING =====
# For numerical calculations, spectral analysis and voice calibration
numpy               # Mathematical operations and arrays
scipy               # Scientific algorithms and DSP
scikit-learn        # Machine learning for voice analysis

# ===== TEXT PROCESSING =====  
# For text cleaning and sentiment analysis
regex               # Advanced regular expressions
pysbd               # Sentence boundary detection
inflect             # Number-to-text conversions
num2words           # Numeric to words conversions

# ===== AUDIO ANALYSIS =====
# For automatic calibration and voice characteristics analysis  
numba               # JIT compilation for performance
joblib              # Parallelization and caching

# ===== UTILITIES =====
# Utilities for progress, logging and configuration
tqdm                # Progress bars for user feedback
PyYAML              # Configuration and settings
requests            # Model and resource downloads
packaging           # Package version management

# ===== HUGGING FACE ECOSYSTEM =====
# For model download and management
huggingface-hub     # Hugging Face Hub client
safetensors         # Safe tensor format

# ===== SYSTEM UTILITIES =====
# For system monitoring and processes  
psutil              # System info and monitoring

# ===== VISUALIZATION (OPTIONAL) =====
# For debugging and analysis (comment out if not using)
matplotlib          # Plotting for audio analysis
# tensorboard       # Metrics visualization (optional)
